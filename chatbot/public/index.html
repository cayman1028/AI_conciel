チャットボット埋め込みデモ

使用方法

以下のコードをあなたのウェブサイトに追加してチャットボットを埋め込みます：

<script src="./chatbot.umd.js"></script>
<script>
  document.addEventListener('DOMContentLoaded', function() {
    initChatBot({
      targetElement: document.getElementById('your-container-id'),
      greeting: 'こんにちは！何かお手伝いできることはありますか？'
    });
  });
</script>
